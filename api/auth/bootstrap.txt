// api/auth/bootstrap.js
import { createClient } from "@supabase/supabase-js";

/**
 * bootstrap endpoint:
 * - Recibe Authorization: Bearer <access_token>
 * - Valida el token contra Supabase Auth (getUser)
 * - Set-Cookie: tg_at=... (HttpOnly)  ✅ fuente de verdad backend
 * - Ejecuta bootstrap_session_context() en contexto de usuario (multi-tenant OK)
 * - Nunca 500 por casos normales (sin env / sin bearer): responde controlado
 */

function isValidSupabaseUrl(url) {
  try {
    const u = new URL(url);
    if (!u.hostname.endsWith(".supabase.co")) return false;
    return true;
  } catch {
    return false;
  }
}

function sameProjectRef(url, expectedRef) {
  try {
    const u = new URL(url);
    const ref = u.hostname.split(".")[0];
    return ref === expectedRef;
  } catch {
    return false;
  }
}

function safeError(err) {
  if (!err) return null;
  return {
    message: err.message,
    code: err.code,
    details: err.details,
    hint: err.hint,
    status: err.status,
  };
}

function parseBearer(req) {
  const h = req?.headers?.authorization || req?.headers?.Authorization || "";
  if (typeof h !== "string") return "";
  const m = h.match(/^Bearer\s+(.+)$/i);
  return m?.[1] || "";
}

function isHttpsRequest(req) {
  // En Vercel Preview/Prod normalmente viene https aquí
  const xfproto = req?.headers?.["x-forwarded-proto"];
  if (xfproto) return String(xfproto).includes("https");
  // fallback conservador
  return true;
}

export default async function handler(req, res) {
  const build_tag = "bootstrap-v1-cookie-sync-preview";

  // CORS/Preflight (por si algún WebView/TWA hace preflight)
  if (req.method === "OPTIONS") {
    res.setHeader("Access-Control-Allow-Origin", "*");
    res.setHeader("Access-Control-Allow-Headers", "content-type, authorization");
    res.setHeader("Access-Control-Allow-Methods", "POST,OPTIONS");
    return res.status(204).end();
  }

  if (req.method !== "POST") {
    return res.status(405).json({ ok: false, build_tag, error_code: "METHOD_NOT_ALLOWED" });
  }

  try {
    // 🔒 Ref fijo del proyecto
    const PROJECT_REF = "mujwsfhkocsuuahlrssn";

    let url = process.env.SUPABASE_URL;
    let anonKey = process.env.SUPABASE_ANON_KEY;

    // fallback controlado (misma lógica que session.js)
    if ((!url || !anonKey) && process.env.VITE_SUPABASE_URL && process.env.VITE_SUPABASE_ANON_KEY) {
      const candidateUrl = process.env.VITE_SUPABASE_URL;
      const candidateAnon = process.env.VITE_SUPABASE_ANON_KEY;

      if (isValidSupabaseUrl(candidateUrl) && sameProjectRef(candidateUrl, PROJECT_REF)) {
        url = url || candidateUrl;
        anonKey = anonKey || candidateAnon;
      }
    }

    if (!url || !anonKey) {
      return res.status(503).json({
        ok: false,
        build_tag,
        error_code: "MISSING_ENV",
        missing: {
          SUPABASE_URL: !process.env.SUPABASE_URL,
          SUPABASE_ANON_KEY: !process.env.SUPABASE_ANON_KEY,
          VITE_SUPABASE_URL: !process.env.VITE_SUPABASE_URL,
          VITE_SUPABASE_ANON_KEY: !process.env.VITE_SUPABASE_ANON_KEY,
        },
      });
    }

    if (!isValidSupabaseUrl(url) || !sameProjectRef(url, PROJECT_REF)) {
      return res.status(503).json({
        ok: false,
        build_tag,
        error_code: "INVALID_SUPABASE_URL",
        url_used: url,
        note: `El SUPABASE_URL debe ser https://${PROJECT_REF}.supabase.co`,
      });
    }

    const access_token = parseBearer(req);
    if (!access_token) {
      return res.status(401).json({
        ok: false,
        build_tag,
        error_code: "MISSING_BEARER",
        note: "Envía Authorization: Bearer <access_token>",
      });
    }

    // Cliente en contexto de usuario
    const sbUser = createClient(url, anonKey, {
      auth: { persistSession: false, autoRefreshToken: false },
      global: { headers: { Authorization: `Bearer ${access_token}` } },
    });

    // 1) Validar token
    const { data: u1, error: uerr1 } = await sbUser.auth.getUser();
    const user = u1?.user ? { id: u1.user.id, email: u1.user.email } : null;

    if (!user || uerr1) {
      return res.status(401).json({
        ok: false,
        build_tag,
        error_code: "INVALID_OR_EXPIRED_TOKEN",
        auth_error: safeError(uerr1),
      });
    }

    // 2) Set-Cookie tg_at (HttpOnly)
    const secure = isHttpsRequest(req);
    const maxAgeSeconds = 60 * 60; // 1 hora (coincide con vida típica del AT)
    const cookieParts = [
      `tg_at=${encodeURIComponent(access_token)}`,
      "HttpOnly",
      "Path=/",
      "SameSite=Lax",
      `Max-Age=${maxAgeSeconds}`,
      secure ? "Secure" : "",
    ].filter(Boolean);

    res.setHeader("Set-Cookie", cookieParts.join("; "));

    // 3) Bootstrap multi-tenant (no rompe nada si falla)
    let boot = null;
    let bootError = null;

    const { data: bootData, error: bootErr } = await sbUser.rpc("bootstrap_session_context");
    if (!bootErr && Array.isArray(bootData) && bootData[0]?.org_id && bootData[0]?.role) {
      boot = { org_id: bootData[0].org_id, role: bootData[0].role };
    } else {
      bootError = safeError(bootErr);
    }

    return res.status(200).json({
      ok: true,
      build_tag,
      authenticated: true,
      cookie_set: true,
      user,
      bootstrapped: !!boot,
      current_org_id: boot?.org_id || null,
      role: boot?.role || null,
      bootstrap_error: bootError,
    });
  } catch (e) {
    console.error("[/api/auth/bootstrap] fatal:", e);
    return res.status(500).json({
      ok: false,
      build_tag,
      error_code: "FATAL",
      error: e?.message || String(e),
    });
  }
}
